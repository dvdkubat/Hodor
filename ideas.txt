function component(width, height, color, name, tag, x, y, speedX, speedY, animation) {
    this.name = name;
    this.tag = tag; 
    this.color = color;
    this.animation = animation;
    this.lastAnimation = 0;
    this.width = width;
    this.height = height;
    this.speedX = speedX;
    this.speedY = speedY;
    this.x = x;
    this.y = y;   
    this.update = function() {
        if(this.alive){ 
          ctx = myGameArea.context;
          if(this.tag == "Player") {
         
            var image = new Image; 
            image.src = this.color;

            ctx = myGameArea.context; 
            ctx.drawImage(image, this.x, this.y, playerWidth, playerHeight);
          }
          else{
            ctx.fillStyle = color;
            ctx.fillRect(this.x, this.y, this.width, this.height);
          }
        }
    } 
}
